145 void DatumConvert(int dIn, double LatIn, double LongIn, double HtIn, int dTo, double& LatTo, double& LongTo, double& HtTo){
146  // converts LatLongHt in datum dIn, to LatLongHt in datum dTo; 2002dec: by Eugene Reimer, from PeterDana equations.
147  // Lat and Long params are in degrees; North latitudes and East longitudes are positive; Height is in meters;
148  // ==This approach to Datum-conversion is a waste of time; to get acceptable accuracy a large table is needed -- see NADCON, NTv2...
149  double a,ee,N,X,Y,Z,EE,p,b,t;
150 
151  //--transform to XYZ, using the "In" ellipsoid
152  //LongIn += 180;
153  a = ellip[datum[dIn].eId].EquatorialRadius;
154  ee= ellip[datum[dIn].eId].eccSquared;
155  N = a / sqrt(1 - ee*sin(LatIn*deg2rad)*sin(LatIn*deg2rad));
156  X = (N + HtIn) * cos(LatIn*deg2rad) * cos(LongIn*deg2rad);
157  Y = (N + HtIn) * cos(LatIn*deg2rad) * sin(LongIn*deg2rad);
158  Z = (N*(1-ee) + HtIn) * sin(LatIn*deg2rad);
159 
160  //--apply delta-terms dX dY dZ
161  //cout<<"\tX:" <<X <<" Y:" <<Y <<" Z:" <<Z; //==DEBUG
162  X+= datum[dIn].dX - datum[dTo].dX;
163  Y+= datum[dIn].dY - datum[dTo].dY;
164  Z+= datum[dIn].dZ - datum[dTo].dZ;
165  //cout<<"\tX:" <<X <<" Y:" <<Y <<" Z:" <<Z; //==DEBUG
166 
167  //--transform back to LatLongHeight, using the "To" ellipsoid
168  a = ellip[datum[dTo].eId].EquatorialRadius;
169  ee= ellip[datum[dTo].eId].eccSquared;
170  EE= ee/(1-ee);
171  p = sqrt(X*X + Y*Y);
172  b = a*sqrt(1-ee);
173  t = atan(Z*a/(p*b));
174  LatTo = atan((Z+EE*b*sin(t)*sin(t)*sin(t)) / (p-ee*a*cos(t)*cos(t)*cos(t)));
175  LongTo= atan(Y/X);
176  HtTo = p/cos(LatTo) - a/sqrt(1-ee*sin(LatTo)*sin(LatTo));
177  LatTo *= rad2deg;
178  LongTo *= rad2deg; LongTo -= 180;
179 }

182 void LLtoUTM(int eId, double Lat, double Long, double& Northing, double& Easting, int& Zone){
183  // converts LatLong to UTM coords; 3/22/95: by ChuckGantz chuck.gantz@globalstar.com, from USGS Bulletin 1532.
184  // Lat and Long are in degrees; North latitudes and East Longitudes are positive.
185  double a = ellip[eId].EquatorialRadius;
186  double ee= ellip[eId].eccSquared;
187  Long -= int((Long+180)/360)*360; //ensure longitude within -180.00..179.9
188  double N, T, C, A, M;
189  double LatRad = Lat*deg2rad;
190  double LongRad = Long*deg2rad;
191 
192  Zone = int((Long + 186)/6);
193  if( Lat >= 56.0 && Lat < 64.0 && Long >= 3.0 && Long < 12.0 ) Zone = 32;
194  if( Lat >= 72.0 && Lat < 84.0 ){ //Special zones for Svalbard
195  if( Long >= 0.0 && Long < 9.0 ) Zone = 31;
196  else if( Long >= 9.0 && Long < 21.0 ) Zone = 33;
197  else if( Long >= 21.0 && Long < 33.0 ) Zone = 35;
198  else if( Long >= 33.0 && Long < 42.0 ) Zone = 37;
199  }
200  double LongOrigin = Zone*6 - 183; //origin in middle of zone
201  double LongOriginRad = LongOrigin * deg2rad;
202 
203  double EE = ee/(1-ee);
204 
205  N = a/sqrt(1-ee*sin(LatRad)*sin(LatRad));
206  T = tan(LatRad)*tan(LatRad);
207  C = EE*cos(LatRad)*cos(LatRad);
208  A = cos(LatRad)*(LongRad-LongOriginRad);
209 
210  M= a*((1 - ee/4 - 3*ee*ee/64 - 5*ee*ee*ee/256 ) *LatRad
211  - (3*ee/8 + 3*ee*ee/32 + 45*ee*ee*ee/1024) *sin(2*LatRad)
212  + (15*ee*ee/256 + 45*ee*ee*ee/1024 ) *sin(4*LatRad)
213  - (35*ee*ee*ee/3072 ) *sin(6*LatRad));
214 
215  Easting = k0*N*(A+(1-T+C)*A*A*A/6+(5-18*T+T*T+72*C-58*EE)*A*A*A*A*A/120) + 500000.0;
216 
217  Northing = k0*(M+N*tan(LatRad)*(A*A/2+(5-T+9*C+4*C*C)*A*A*A*A/24
218  + (61-58*T+T*T+600*C-330*EE)*A*A*A*A*A*A/720));
219 }

222 void UTMtoLL(int eId, double Northing, double Easting, int Zone, double& Lat, double& Long){
223  // converts UTM coords to LatLong; 3/22/95: by ChuckGantz chuck.gantz@globalstar.com, from USGS Bulletin 1532.
224  // Lat and Long are in degrees; North latitudes and East Longitudes are positive.
225  double a = ellip[eId].EquatorialRadius;
226  double ee = ellip[eId].eccSquared;
227  double EE = ee/(1-ee);
228  double e1 = (1-sqrt(1-ee))/(1+sqrt(1-ee));
229  double N1, T1, C1, R1, D, M, mu, phi1Rad;
230  double x = Easting - 500000.0; //remove 500,000 meter offset for longitude
231  double y = Northing;
232  double LongOrigin = Zone*6 - 183; //origin in middle of zone
233 
234  M = y / k0;
235  mu = M/(a*(1-ee/4-3*ee*ee/64-5*ee*ee*ee/256));
236 
237  phi1Rad = mu + (3*e1/2-27*e1*e1*e1/32) *sin(2*mu)
238  + (21*e1*e1/16-55*e1*e1*e1*e1/32) *sin(4*mu)
239  + (151*e1*e1*e1/96) *sin(6*mu);
240  N1 = a/sqrt(1-ee*sin(phi1Rad)*sin(phi1Rad));
241  T1 = tan(phi1Rad)*tan(phi1Rad);
242  C1 = EE*cos(phi1Rad)*cos(phi1Rad);
243  R1 = a*(1-ee)/pow(1-ee*sin(phi1Rad)*sin(phi1Rad), 1.5);
244  D = x/(N1*k0);
245 
246  Lat = phi1Rad - (N1*tan(phi1Rad)/R1)*(D*D/2-(5+3*T1+10*C1-4*C1*C1-9*EE)*D*D*D*D/24
247  +(61+90*T1+298*C1+45*T1*T1-252*EE-3*C1*C1)*D*D*D*D*D*D/720);
248  Lat *= rad2deg;
249  Long = (D-(1+2*T1+C1)*D*D*D/6+(5-2*C1+28*T1-3*C1*C1+8*EE+24*T1*T1)*D*D*D*D*D/120) / cos(phi1Rad);
250  Long = LongOrigin + Long*rad2deg;
251 }
